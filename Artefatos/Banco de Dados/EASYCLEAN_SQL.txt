-- CREATE DATABASE EASYCLEAN

USE EASYCLEAN;



CREATE TABLE DIARISTA

(

  ID_DIARISTA DECIMAL(5,0),

  CPF VARCHAR(20),

  EMAIL VARCHAR(100),

  IMG_DIARISTA VARCHAR(100),

  TELEFONE DECIMAL (14, 0),

  NOME VARCHAR(50),

  CONSTRAINT PK CPF PRIMARY KEY (ID_DIARISTA)

  CONSTRAINT PK ID_DIARISTA

  PRIMARY KEY (ID_DIARISTA)


);



CREATE TABLE LOGRADOURO

(
  
CONSTRAINT FK_ID_DIARISTA FOREIGN KEY (ID_DIARISTA) REFERENCES DIARISTA(ID_DIARISTA),

CONSTRAINT FK_ID_CLIENTE FOREIGN KEY (ID_CLIENTE) REFERENCES CLIENTE(ID_CLIENTE),


);



CREATE TABLE DISPONIBILIDADE

(
   ID_DISPONIBILIDADE DECIMAL(5,0),
  DESCRICAO VARCHAR(500),
  DIA_SEMANA DECIMAL(5,0),
  TELEFONE DECIMAL (14, 0),
  CONSTRAINT PK ID_DISPONIBILIDADE PRIMARY KEY (ID_DISPONIBILIDADE)


);


CREATE TABLE CLIENTE

(

 ID_DISPONIBILIDADE DECIMAL(5,0),
  DESCRICAO VARCHAR(500),
  DIA_SEMANA DECIMAL(5,0),
  TELEFONE DECIMAL (14, 0),
  CONSTRAINT PK ID_DISPONIBILIDADE PRIMARY KEY (ID_DISPONIBILIDADE)


);

CREATE TABLE CONTRATO{

 ID_CONTRATO INT(10) NOT NULL,

 ID_DISPONIBILIDADE(10) NOT NULL,

 NOTA CLIENTE INT(2) NOT NULL,

 NOTA DIARISTA INT(2) NOT NULL,

 VALOR MONEY NOT NULL, 

 FORMA PAGAMENTO CHAR(15) DEFAULT ‘Dinheiro’,

 CONSTRAINT PK_IDContrato PRIMARY KEY (ID_CONTRATO),

 CONSTRAINT FK_IDDisponiblidade FOREIGN KEY (ID_DISPONIBILIDADE) REFERENCES DISPONIBILIDADE(ID_DISPONIBILIDADE)

}
